## encodedTx

Get transaction raw

### Usage

```typescript
import { aelf } from '@portkey/utils';

const raw = await aelf.encodedTx(params as EncodedParams)
```

### Supplemental Code

```typescript
type EncodedParams = {
  instance: any;
  functionName: string;
  paramsOption: any;
  contract: any;
};
```

## getELFContract

Get elf contract

### Usage

```typescript
import { aelf } from '@portkey/utils';

const contract = await aelf.getELFContract(rpcUrl, tokenAddress, privateKey)
```

### Supplemental Code

```typescript
parmas: {
  rpcUrl: string; 
  tokenAddress: string;
  privateKey: string;
}
```

## handleVerificationDoc

Get guardian detail by verificationDoc

### Usage

```typescript
import { aelf } from '@portkey/utils';

const detail = await aelf.handleVerificationDoc(verificationDoc)
```

### Supplemental Code

```typescript
detail
{
  type, // guardian type
  guardianIdentifier, 
  verificationTime, 
  verifierAddress, 
  salt,
}
```

### Example response
```typescript
{ 
  type: 'Email', 
  guardianIdentifier: 'fsdf...grger', 
  verificationTime: 'xxxx', 
  verifierAddress: 'http://xxx.com', 
  salt: "1c02...7631",
}
```
